#include<stdio.h>
int main()
{
int arr[100];
int n,a,i,j;
printf("ENTER THE NUMBER OF ELEMENTS YOU WANT TO HAVE IN ARRAY:\n");
scanf("%d",&n);
printf("ENTER THE ARRAY:\n");
for(a=0;a<n;a++)
{
scanf("%d",&arr[a]);
}
printf("YOUR ENTERED ARRAY IS:\n");
for(a=0;a<n;a++)
{
printf("%d ",arr[a]);
}
for(i=0;i<n-1;i++)
{
	for(j=0;j<n-1;j++)
	{
	  if(arr[j]>arr[j+1])
	  {
	    arr[j]=arr[j]+arr[j+1];
	    arr[j+1]=arr[j]-arr[j+1];
	    arr[j]=arr[j]-arr[j+1];
	  }
	}
}
printf("YOUR SORTED ARRAY IS:\n");
for(a=0;a<n;a++)
{
printf("%d ",arr[a]);
}
return 0;
}
