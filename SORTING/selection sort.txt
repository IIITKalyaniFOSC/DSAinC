#include<stdio.h>
int main()
{
int arr[100];
int n,a,i,j,min;
printf("ENTER THE NUMBER OF ELEMENTS YOU WANT TO HAVE IN ARRAY:\n");
scanf("%d",&n);
printf("ENTER THE ARRAY:\n");
for(a=0;a<n;a++)
{
scanf("%d",&arr[a]);
}
printf("YOUR ENTERED ARRAY IS:\n");
for(a=0;a<n;a++)
{
printf("%d ",arr[a]);
}
for(i=0;i<n-1;i++)
{
	min=arr[0];
	for(j=i;j<n;j++)
	{
	  if(arr[j]<min)
	  {
	    min=arr[j];
	    arr[j]=arr[i];
	    arr[i]=min;
	  }
	}
}
printf("YOUR SORTED ARRAY IS:\n");
for(a=0;a<n;a++)
{
printf("%d ",arr[a]);
}
return 0;
}
